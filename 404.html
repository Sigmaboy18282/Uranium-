const form = document.getElementById('proxy-form'); const input = document.getElementById('url-input'); form.addEventListener('submit', async (e) => { e.preventDefault(); // Register the worker using the Repo Name path await navigator.serviceWorker.register('/Uranium-/sw.js', { scope: '/Uranium-/service/' }); let url = input.value.trim(); if (!url.includes('.') || url.includes(' ')) { url = 'https://www.google.com/search?q=' + encodeURIComponent(url); } else if (!url.startsWith('http')) { url = 'https://' + url; } // Redirect to the service path window.location.href = '/Uranium-/service/' + btoa(url); });
